Two ways to create and edit `.bash_profile` in OSX.

{% highlight bash %}
cd ~/
touch .bash_profile
nano .bash_profile
{% endhighlight %}

{% highlight bash %}
cd ~/
touch .bash_profile
open -e .bash_profile
{% endhighlight %}
